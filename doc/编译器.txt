编译器将源代码翻译为目标代码的过程叫做编译（compiler）

// 源代码
<template>
   <div @click="clickChange">
      你好呀，鱼泡
   </div>
</template>

// 目标代码
function render () {
  return h('div', {onClick: clickChange }, '你好呀，鱼泡')
}
